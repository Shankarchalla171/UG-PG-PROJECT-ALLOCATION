\section{Introduction}

% Slide 3: Problem Statement
\begin{frame}{Problem Statement}
    \begin{itemize}
        \item \textbf{Current Process:} Departments at NIT Calicut rely on manual, paper-based project allocation.
        \item \textbf{Challenges:}
        \begin{itemize}
            \item Prone to miscommunication and lost records.
            \item Inequitable distribution of faculty workload.
            \item High administrative overhead every semester.
            \item Lack of real-time tracking for students and faculty.
        \end{itemize}
    \end{itemize}
\end{frame}

% Slide 4: Proposed Solution
\begin{frame}{Proposed Solution: Edu Project}
    \begin{itemize}
        \item A centralized, college-internal web portal to digitalize the UG/PG project lifecycle.
        \item \textbf{Key Capabilities:}
        \begin{itemize}
            \item Students browse projects and register preferences.
            \item Faculty propose and manage project titles dynamically.
            \item Department Coordinators oversee fair, conflict-free allocation.
            \item Real-time tracking and automated notifications.
        \end{itemize}
    \end{itemize}
\end{frame}

% Slide 5: User Roles
\begin{frame}{User Roles \& Actors}
    \begin{itemize}
        \item \textbf{Student:} Browse projects, form teams, apply as Team Leader, track status.
        \item \textbf{Faculty (Professor):} Propose projects, define domains/slots, review applications, accept/reject teams.
        \item \textbf{Department Coordinator:} Set faculty intake limits, manage phases (Team Formation, Allocation), resolve conflicts.
        \item \textbf{Admin:} Create users, assign system roles, system governance.
    \end{itemize}
\end{frame}

% Slide 6: Key Features
\begin{frame}{Core System Features}
    \begin{itemize}
        \item Role-based authentication (JWT \& Google OAuth).
        \item Faculty-driven project catalogue with capacity limits.
        \item Dynamic slot management and capacity control.
        \item Multi-semester project duration handling.
        \item Automated audit logs and activity tracking.
    \end{itemize}
\end{frame}

\section{Core Workflows}

% Slide 11: Team Formation Flow
\begin{frame}{Workflow: Team Formation}
    \begin{enumerate}
        \item \textbf{Create Team:} A student creates a team and becomes the Team Leader (max 3 members).
        \item \textbf{Generate ID:} A unique Team ID is generated.
        \item \textbf{Join Team:} Other members join using the Team ID.
        \item \textbf{Finalize:} Team is locked before applying to projects.
    \end{enumerate}
\end{frame}

% Slide 12: 3-Way Handshake Allocation
\begin{frame}{Workflow: 3-Way Handshake Allocation}
    \begin{enumerate}
        \item \textbf{Apply to Project (Team Leader):} Submits application with resume $\rightarrow$ \texttt{PENDING}.
        \item \textbf{Professor Decision:} Reviews application and accepts/rejects $\rightarrow$ \texttt{CONFIRMED} or \texttt{REJECTED}.
        \item \textbf{Team Confirmation:} Team Leader accepts the offer $\rightarrow$ \texttt{TEAM\_CONFIRMED}.
        \item \textbf{Finalization:} Project is officially assigned; other pending applications for the team are auto-rejected.
    \end{enumerate}
\end{frame}

% Slide 13: Phase Management
\begin{frame}{Phase Management (Coordinator)}
    \begin{itemize}
        \item Coordinators enforce strict deadlines for different phases:
        \begin{enumerate}
            \item \textbf{Team Formation Phase:} Students group up.
            \item \textbf{Project Creation Phase:} Faculty propose topics and slots.
            \item \textbf{Project Allocation Phase:} 3-way handshake execution.
        \end{enumerate}
        \item The system restricts actions outside of active phases.
    \end{itemize}
\end{frame}

% Slide 14: Database Design
\begin{frame}{Database Entities (Overview)}
    \begin{itemize}
        \item \textbf{User:} id, name, email, role, dept, passwordHash.
        \item \textbf{Project:} title, description, capacity, professor, phase, projectDuration.
        \item \textbf{Team:} teamId, leader, members, status.
        \item \textbf{ProjectApplications:} team, project, status (PENDING $\rightarrow$ CONFIRMED).
    \end{itemize}
\end{frame}
